Estruturas condicionais

IF-ELSE STATEMENT

if(/*condition*/){
	//code
}else if(/*condition*/){
	//code
}else{
	//code
}


SWITCH STATEMENT

switch(expression){
   case constant-expression  :
      statement(s);
      break; //optional
   case constant-expression  :
      statement(s);
      break; //optional
  
   // you can have any number of case statements.
   default : //Optional
      statement(s);
}

Casos especiais

Os problemas de maratona de computação requerem um tratamento especial das entradas de testes do algoritmos.

  • Não é necessário informar ao 'usuário' para informar a entrada de dados. 
    INCORRETO
    
#include <iostream>

using namespace std;

int main() {
	int idade;
	cout << "Digite sua idade: " << endl;
	cin >> idade;
  	cout << "Sua idade é: " << idade << endl;
  	return 0;
}


    CORRETO
    
#include <iostream>

using namespace std;

int main() {
	int idade;
	cin >> idade;
  	cout << idade << endl;
  	return 0;
}
  • Muitas vezes é preciso testar diversas entradas para a mesma execução do algorítmo. 
    Nº DE EXECUÇÕES DEFINIDAS
    
#include <iostream>
#include <math.h>
using namespace std;

int main() {
	int n;
	float raiz_do_numero;
	cin >> idade;
	for(int i = 0; i < n; i++){
		cin >> raiz_do_numero;
		cout << sqrt(raiz_do_numero) << endl;
	}
  	return 0;
}


    Nº DE EXECUÇÕES INDEFINIDAS
    
#include <iostream>
#include <cmath>
using namespace std;

int main() {
	float raiz_do_numero;
	while(cin >> raiz_numero){
		cout << sqrt(raiz_do_numero) << endl;
	}
  	return 0;
}
